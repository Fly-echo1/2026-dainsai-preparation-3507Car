# 自动驾驶小车 — 实现计划

> **For agentic workers:** 使用 superpowers:executing-plans 按 task 顺序实现。每个 task 的步骤使用 `- [ ]` checkbox 跟踪。

**目标：** 基于 MSPM0G3507 平台，从零实现 2024 电赛 H 题 4 个任务的自动驾驶小车固件。

**架构：** 阻塞式状态机，每个路段封装为阻塞函数（`Control_StraightDrive` / `Control_ArcFollow` / `Control_TurnToAngle`），flag 状态位管理路段切换，差异化判据检测站点。

**技术栈：** TI MSPM0G3507, CCS Theia, SysConfig, DriverLib, C

## 全局约束

- 所有 .c/.h 文件使用 UTF-8 编码，中文注释
- 驱动层函数只依赖 `ti_msp_dl_config.h` 和本模块 .h
- 控制层不直接操作寄存器，通过驱动层 API 访问硬件
- 关键参数集中在 `config.h`，用 `#define` 宏定义
- 无动态内存分配，所有变量为全局或 static

```c
#include "pid.h"
#include <stddef.h>

void PID_Init(PID_t *pid, float Kp, float Ki, float Kd)
{
    pid->Kp = Kp;
    pid->Ki = Ki;
    pid->Kd = Kd;
    pid->error = 0.0f;
    pid->last_error = 0.0f;
    pid->last2_error = 0.0f;
    pid->output = 0.0f;
    pid->out_min = -1e6f;
    pid->out_max = 1e6f;
}

void PID_SetOutputLimit(PID_t *pid, float min, float max)
{
    pid->out_min = min;
    pid->out_max = max;
}

void PID_Reset(PID_t *pid)
{
    pid->error = 0.0f;
    pid->last_error = 0.0f;
    pid->last2_error = 0.0f;
    pid->output = 0.0f;
}

float PID_Calc(PID_t *pid, float error)
{
    pid->error = error;
    // 增量式: Δu = Kp*(e - e1) + Ki*e + Kd*(e - 2*e1 + e2)
    pid->output += pid->Kp * (error - pid->last_error)
                 + pid->Ki * error
                 + pid->Kd * (error - 2.0f * pid->last_error + pid->last2_error);

    pid->last2_error = pid->last_error;
    pid->last_error = error;

    // 输出限幅
    if (pid->output > pid->out_max) pid->output = pid->out_max;
    if (pid->output < pid->out_min) pid->output = pid->out_min;

    return pid->output;
}

float PID_CalcAngle180(PID_t *pid, float target, float current)
{
    // 角度误差归一化到 (-180, 180]
    float error = target - current;
    while (error > 180.0f)  error -= 360.0f;
    while (error < -180.0f) error += 360.0f;
    return PID_Calc(pid, error);
}
```

```c
/*
 * 站点检测：不同 task 用不同判据，通过 g_state.task 和 g_state.flag 区分。
 * 
 * Task 1: B点 = 任一路黑线 ×30
 * Task 2: B点 = 任一路黑线 ×30, C点 = 全白 ×5000, D点 = 右侧4路黑 ×30, A点 = 全白 ×7000
 * Task 3/4: C点 = 左侧4路黑 ×50, B点 = 全白 ×7000, D点 = 右侧4路黑 ×50, A点 = 全白 ×7000
 */
bool Control_IsStation(void) {
    uint8_t sensors[GRAYSCALE_CHANNELS];
    Grayscale_Sensor_Read_All(sensors);

    uint8_t task = (uint8_t)g_state.task;
    uint8_t flag = g_state.flag;
    uint8_t threshold, total_black, right_black, left_black;
    total_black = count_black(sensors);
    right_black = count_right_black(sensors);
    left_black = count_left_black(sensors);

    bool condition = false;

    switch (task) {
    case 1: /* Task 1: B点 */
        condition = (total_black > 0);
        threshold = STATION_DEBOUNCE_BLACK_T1;
        break;

    case 2: /* Task 2 */
        if (flag == 1)      { /* B点 */ condition = (total_black > 0); threshold = STATION_DEBOUNCE_BLACK_T2; }
        else if (flag == 2) { /* C点 */ condition = (total_black == 0); threshold = STATION_DEBOUNCE_WHITE_T2; }
        else if (flag == 3) { /* D点 */ condition = (right_black > 0);  threshold = STATION_DEBOUNCE_BLACK_T2; }
        else if (flag == 4) { /* A点 */ condition = (total_black == 0); threshold = STATION_DEBOUNCE_WHITE_T3; }
        break;

    case 3: /* fallthrough */
    case 4: /* Task 3/4 */
        if (flag == 1)      { /* C点 */ condition = (left_black > 0);   threshold = STATION_DEBOUNCE_BLACK_T3; }
        else if (flag == 2) { /* B点 */ condition = (total_black == 0); threshold = STATION_DEBOUNCE_WHITE_T3; }
        else if (flag == 3) { /* D点 */ condition = (right_black > 0);  threshold = STATION_DEBOUNCE_BLACK_T3; }
        else if (flag == 4) { /* A点 */ condition = (total_black == 0); threshold = STATION_DEBOUNCE_WHITE_T3; }
        break;
    }

    /* 去抖 */
    if (condition) {
        g_station_count++;
        if (g_station_count >= threshold) { g_station_count = 0; return true; }
    } else {
        g_station_count = 0;
    }
    return false;
}
```

```c
bool Control_StraightDrive(float target_yaw_deg, float target_distance_m) {
    g_distance_m = 0.0f;
    PID_Reset(&pid_angle);
    PID_Reset(&pid_speed);

    while (g_distance_m < target_distance_m) {
        /* 站点检测 */
        if (Control_IsStation()) { Motor_Stop(); return true; }

        /* 直线段不检测脱线 — 直接继续 */

        /* 角度 PID */
        float yaw = wit_data.yaw;
        float steer = PID_CalcAngle180(&pid_angle, target_yaw_deg, yaw);

        /* 速度 PID */
        float avg_spd = (encoder_left_speed + encoder_right_speed) / 2.0f;
        float spd_out = PID_Calc(&pid_speed, (float)g_base_speed - avg_spd);

        /* 合成输出 */
        int16_t left  = (int16_t)(g_base_speed + spd_out + steer);
        int16_t right = (int16_t)(g_base_speed + spd_out - steer);

        if (left  > 100) left  = 100; if (left  < -100) left  = -100;
        if (right > 100) right = 100; if (right < -100) right = -100;

        Motor_SetSpeed(left, right);
        update_distance();
        mspm0_delay_ms(10);
    }

    Motor_Stop();
    return false;
}
```

```c
bool Control_ArcFollow(float distance_threshold_m) {
    g_distance_m = 0.0f;
    PID_Reset(&pid_omega);
    static uint8_t offtrack_count = 0;

    while (g_distance_m < distance_threshold_m) {
        /* 站点检测 */
        if (Control_IsStation()) { Motor_Stop(); return true; }

        /* 脱线检测 — 减速继续 */
        if (Control_IsOffTrack()) {
            offtrack_count++;
            if (offtrack_count >= OFFTRACK_DEBOUNCE) {
                int16_t slow = ARC_SPEED / 2;
                if (slow < 10) slow = 10;
                Motor_SetSpeed(slow, slow);
            }
        } else {
            offtrack_count = 0;
        }

        /* 灰度查表 */
        uint8_t sensors[GRAYSCALE_CHANNELS];
        Grayscale_Sensor_Read_All(sensors);
        int8_t idx = find_first_black(sensors);
        float target_omega = (idx >= 0 && idx < GRAYSCALE_CHANNELS)
                             ? SENSOR_TO_OMEGA[idx] : 0.0f;

        /* 角速度 PID */
        float current_omega = wit_data.gz;
        float omega_out = PID_Calc(&pid_omega, target_omega - current_omega);

        /* 差速输出 */
        int16_t left  = (int16_t)(ARC_SPEED + omega_out);
        int16_t right = (int16_t)(ARC_SPEED - omega_out);

        if (left  > 100) left  = 100; if (left  < -100) left  = -100;
        if (right > 100) right = 100; if (right < -100) right = -100;

        Motor_SetSpeed(left, right);
        update_distance();
        mspm0_delay_ms(10);
    }

    Motor_Stop();
    return false;
}
```

```c
bool Control_TurnToAngle(float target_yaw_deg) {
    int16_t ts = TURN_SPEED;

    while (1) {
        float error = target_yaw_deg - wit_data.yaw;
        while (error > 180.0f)  error -= 360.0f;
        while (error < -180.0f) error += 360.0f;

        if (error < ANGLE_THRESHOLD && error > -ANGLE_THRESHOLD) {
            Motor_Stop();
            return true;
        }

        if (error > 0) Motor_SetSpeed(-ts, ts);
        else           Motor_SetSpeed(ts, -ts);

        mspm0_delay_ms(5);
    }
}
```

```c
void Task_Run(void) {
    float start_yaw = wit_data.yaw;
    uint32_t start_tick = tick_ms;
    float timeout, elapsed;
    bool seg_result;

    Control_H_Init();
    Control_ResetDistance();
    g_state.flag = 1;
    g_state.lap = 0;
    g_state.is_running = true;
    g_state.is_timeout = false;
    g_state.is_offtrack = false;

    switch (g_state.task) {
        case TASK_1: timeout = TASK1_TIMEOUT; break;
        case TASK_2: timeout = TASK2_TIMEOUT; break;
        case TASK_3: timeout = TASK3_TIMEOUT; break;
        case TASK_4: timeout = TASK4_TIMEOUT; break;
        default: return;
    }

    BT_SendLog("{\"event\":\"task_start\",\"task\":%d,\"yaw\":%.1f}\r\n",
               (int)g_state.task, start_yaw);

    /* ===== Task 1: A→B ===== */
    if (g_state.task == TASK_1) {
        g_state.flag = 1;
        Control_SetBaseSpeed(BASE_SPEED);

        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"AB\"}\r\n");
        seg_result = Control_StraightDrive(start_yaw, STRAIGHT_DIST);
        elapsed = (tick_ms - start_tick) / 1000.0f;

        if (elapsed > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(1);  /* B */
        BT_SendLog("{\"event\":\"seg_done\",\"seg\":\"AB\",\"result\":%s}\r\n",
                   seg_result ? "true" : "false");
        goto task_end;
    }

    /* ===== Task 2: A→B→C→D→A ===== */
    if (g_state.task == TASK_2) {
        /* AB */
        g_state.flag = 1;
        Control_SetBaseSpeed(BASE_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"AB\"}\r\n");
        seg_result = Control_StraightDrive(start_yaw, STRAIGHT_DIST);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(1);

        /* BC */
        g_state.flag = 2;
        Control_SetBaseSpeed(ARC_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"BC\"}\r\n");
        seg_result = Control_ArcFollow(ARC_BC_DIST);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(2);

        /* CD */
        g_state.flag = 3;
        Control_SetBaseSpeed(BASE_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"CD\"}\r\n");
        seg_result = Control_StraightDrive(start_yaw + 180.0f, STRAIGHT_DIST);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(3);

        /* DA */
        g_state.flag = 4;
        Control_SetBaseSpeed(ARC_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"DA\"}\r\n");
        seg_result = Control_ArcFollow(ARC_DA_DIST_T2);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(0);

        goto task_end;
    }

    /* ===== Task 3: A→C→B→D→A (交叉) ===== */
    if (g_state.task == TASK_3) {
        float turn_yaw;

        /* A→C */
        g_state.flag = 1;
        Control_SetBaseSpeed(BASE_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"AC\"}\r\n");
        seg_result = Control_StraightDrive(start_yaw, STRAIGHT_DIST * 1.5f);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(2);  /* C */

        /* 原地转弯进入 CB */
        turn_yaw = wit_data.yaw + TURN_ANGLE_90;
        Control_SetBaseSpeed(TURN_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"TURN\"}\r\n");
        Control_TurnToAngle(turn_yaw);

        /* C→B 弧线 */
        g_state.flag = 2;
        Control_SetBaseSpeed(ARC_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"CB\"}\r\n");
        seg_result = Control_ArcFollow(ARC_CB_DIST);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(1);  /* B */

        /* B→D 直线 */
        g_state.flag = 3;
        Control_SetBaseSpeed(BASE_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"BD\"}\r\n");
        seg_result = Control_StraightDrive(turn_yaw, STRAIGHT_DIST);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(3);  /* D */

        /* D→A 弧线 */
        g_state.flag = 4;
        Control_SetBaseSpeed(ARC_SPEED);
        BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"DA\"}\r\n");
        seg_result = Control_ArcFollow(ARC_DA_DIST_T3);
        if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
        Control_StationAlert(0);  /* A */

        goto task_end;
    }

    /* ===== Task 4: ×4 圈 ===== */
    if (g_state.task == TASK_4) {
        for (g_state.lap = 1; g_state.lap <= 4; g_state.lap++) {
            float turn_yaw;
            float lap_comp = (g_state.lap - 1) * 0.15f;

            /* A→C */
            g_state.flag = 1;
            Control_SetBaseSpeed(BASE_SPEED);
            BT_SendLog("{\"event\":\"seg_start\",\"seg\":\"AC\",\"lap\":%d}\r\n", (int)g_state.lap);
            seg_result = Control_StraightDrive(start_yaw, STRAIGHT_DIST * 1.5f);
            if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
            Control_StationAlert(2);

            /* 转弯 */
            turn_yaw = wit_data.yaw + TURN_ANGLE_90;
            Control_SetBaseSpeed(TURN_SPEED);
            Control_TurnToAngle(turn_yaw);

            /* C→B 弧线 */
            g_state.flag = 2;
            Control_SetBaseSpeed(ARC_SPEED);
            seg_result = Control_ArcFollow(ARC_CB_DIST + lap_comp);
            if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
            Control_StationAlert(1);

            /* B→D 直线 */
            g_state.flag = 3;
            Control_SetBaseSpeed(BASE_SPEED);
            seg_result = Control_StraightDrive(turn_yaw, STRAIGHT_DIST);
            if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
            Control_StationAlert(3);

            /* D→A 弧线 */
            g_state.flag = 4;
            Control_SetBaseSpeed(ARC_SPEED);
            seg_result = Control_ArcFollow(ARC_DA_DIST_T3 + lap_comp);
            if ((tick_ms - start_tick) / 1000.0f > timeout) { g_state.is_timeout = true; goto task_end; }
            Control_StationAlert(0);

            /* 每圈结束: 非最后一圈掉头 */
            if (g_state.lap < 4) {
                Control_SetBaseSpeed(TURN_SPEED);
                Control_TurnToAngle(wit_data.yaw + 180.0f);
                Control_ResetDistance();
            }

            BT_SendLog("{\"event\":\"lap_done\",\"lap\":%d,\"time\":%.1f}\r\n",
                       (int)g_state.lap, (tick_ms - start_tick) / 1000.0f);
        }
        goto task_end;
    }

task_end:
    Motor_Stop();
    g_state.is_running = false;
    g_state.elapsed_s = (tick_ms - start_tick) / 1000.0f;
    g_state.distance_m = Control_GetDistance();

    BT_SendLog("{\"event\":\"task_end\",\"task\":%d,\"timeout\":%s,\"dist\":%.2f,\"time\":%.1f}\r\n",
               (int)g_state.task, g_state.is_timeout ? "true" : "false",
               g_state.distance_m, g_state.elapsed_s);

    /* 结束蜂鸣 */
    Buzzer_BeepLong();
}
```
